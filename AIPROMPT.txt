async function syncToServer() {
  if (!confirm('Send all clearance data to central server? This may contain PII.')) return;

  try {
    const data = await fetchAllData();  // your existing function
    if (data.length === 0) {
      alert('No data to sync.');
      return;
    }

    // Optional: Show progress
    const btn = document.getElementById('syncToServerBtn');
    btn.disabled = true;
    btn.textContent = 'Uploading...';

    const response = await fetch('https://your-server.com/api/clearance/import', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        // 'Authorization': 'Bearer ' + yourTokenIfNeeded
      },
      body: JSON.stringify({
        uploadedBy: document.getElementById('processedBy').value.trim() || 'Unknown',
        uploadDate: new Date().toISOString(),
        records: data
      })
    });

    if (response.ok) {
      const result = await response.json();
      alert(`Success! ${result.inserted || data.length} records saved to central database.`);
    } else {
      throw new Error(`Server error: ${response.status}`);
    }
  } catch (err) {
    console.error(err);
    alert('Sync failed: ' + err.message);
  } finally {
    const btn = document.getElementById('syncToServerBtn');
    btn.disabled = false;
    btn.innerHTML = '<i class="bi bi-cloud-upload"></i> Sync to Server (SQLite)';
  }
}

// Attach it
document.getElementById('syncToServerBtn')?.addEventListener('click', syncToServer);
